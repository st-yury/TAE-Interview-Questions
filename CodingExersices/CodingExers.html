class Program
{
  public static void Main()
  {
      List<string> listOfNamesWithDuplicates = new()
      {
          "Alice", "Bob", "Charlie",
          "David", "Emma", "Frank",
          "Grace", "Henry", "Irene",
          "Jack", "Kate", "Leo",
          "Mary", "Noah", "Olivia",
          "Peter", "Quinn", "Rachel",
          "Sam", "Sarah", "Tom",
          "Alice", "Frank", "Irene",
          "David", "Grace", "Jack",
          "Mary", "Noah", "Noah", "Jack"
      };

      List<string> onlyUniqueNames = new();

      foreach (var name in listOfNamesWithDuplicates)
      {
          if (!onlyUniqueNames.Contains(name))
          {
              onlyUniqueNames.Add(name);
          }
      }

      Console.WriteLine($"Only unique names: {string.Join(", ", onlyUniqueNames)}");

      // Only unique names: Alice, Bob, Charlie, David, Emma, Frank, Grace,
      // Henry, Irene, Jack, Kate, Leo, Mary, Noah, Olivia, Peter, Quinn, Rachel, Sam, Sarah, Tom
  }
}
