A way of passing a request between a chain of objects.
// Handler interface declares a method for handling requests and a reference to the next handler.
interface IHandler
{
    void HandleRequest(int request);
    void SetNextHandler(IHandler nextHandler);
}

// ConcreteHandlerA and ConcreteHandlerB are concrete implementations of the handler interface.
class ConcreteHandlerA : IHandler
{
    private IHandler nextHandler;

    public void SetNextHandler(IHandler nextHandler)
    {
        this.nextHandler = nextHandler;
    }

    public void HandleRequest(int request)
    {
        if (request <= 10)
        {
            Console.WriteLine($"Handler A can handle the request {request}");
        }
        else if (nextHandler != null)
        {
            Console.WriteLine($"Handler A cannot handle the request {request}. Passing to the next handler.");
            nextHandler.HandleRequest(request);
        }
        else
        {
            Console.WriteLine($"No handler can process the request {request}");
        }
    }
}

class ConcreteHandlerB : IHandler
{
    private IHandler nextHandler;

    public void SetNextHandler(IHandler nextHandler)
    {
        this.nextHandler = nextHandler;
    }

    public void HandleRequest(int request)
    {
        if (request > 10 && request <= 20)
        {
            Console.WriteLine($"Handler B can handle the request {request}");
        }
        else if (nextHandler != null)
        {
            Console.WriteLine($"Handler B cannot handle the request {request}. Passing to the next handler.");
            nextHandler.HandleRequest(request);
        }
        else
        {
            Console.WriteLine($"No handler can process the request {request}");
        }
    }
}

// Client code creates a chain of handlers and initiates the request.
class Client
{
    private IHandler chain;

    public void BuildChain()
    {
        IHandler handlerA = new ConcreteHandlerA();
        IHandler handlerB = new ConcreteHandlerB();

        handlerA.SetNextHandler(handlerB);

        chain = handlerA;
    }

    public void MakeRequest(int request)
    {
        chain.HandleRequest(request);
    }
}

class Program
{
    static void Main(string[] args)
    {
        // Client code creates a chain of handlers and initiates a request.
        Client client = new();
        client.BuildChain();

        // Requests with different values to demonstrate the chain of responsibility.
        client.MakeRequest(5);
        client.MakeRequest(15);
        client.MakeRequest(25);

        // Handler A cannot handle the request 15.Passing to the next handler.
        // Handler B can handle the request 15
        // Handler A cannot handle the request 25.Passing to the next handler.
        // No handler can process the request 25
    }
}

    }

// NonTerminalExpression is a concrete expression that interprets non-terminals in the grammar.
class NonTerminalExpression : IAbstractExpression

    private IAbstractExpression expression2;

    public NonTerminalExpression(IAbstractExpression expression1, IAbstractExpression expression2)
    {
        this.expression1 = expression1;
        this.expression2 = expression2;
    }

    public void Interpret(Context context)
    {
        Console.WriteLine("NonTerminalExpression interprets based on context:");
        expression1.Interpret(context);
        expression2.Interpret(context);
    }
}

// Context contains information that's being interpreted.
class Context
{
    public int Output { get; set; }
}

Sequentially access the elements of an aggregate object.
// Aggregate interface declares an Iterator method for creating an iterator.
interface IAggregate
{
    IIterator CreateIterator();
}

// Iterator interface declares methods for traversing the collection.
interface IIterator
{
    object Current();
    bool MoveNext();
    void Reset();
}

// ConcreteAggregate is a concrete implementation of the Aggregate interface.
class ConcreteAggregate : IAggregate
{
    private object[] items = { "Item 1", "Item 2", "Item 3", "Item 4", "Item 5" };

    public IIterator CreateIterator()
    {
        return new ConcreteIterator(this);
    }

    public int Count
    {
        get { return items.Length; }
    }

    public object this[int index]
    {
        get { return items[index]; }
    }
}

// ConcreteIterator is a concrete implementation of the Iterator interface.
class ConcreteIterator : IIterator
{
    private ConcreteAggregate aggregate;
    private int currentIndex = 0;

    public ConcreteIterator(ConcreteAggregate aggregate)
    {
        this.aggregate = aggregate;
    }

    public object Current()
    {
        return aggregate[currentIndex];
    }

    public bool MoveNext()
    {
        currentIndex++;
        return currentIndex < aggregate.Count;
    }

    public void Reset()
    {
        currentIndex = 0;
    }
}

class Program
{
    static void Main(string[] args)
    {
        // Client code creates an iterator and traverses the collection.
        ConcreteAggregate aggregate = new ConcreteAggregate();
        IIterator iterator = aggregate.CreateIterator();

        while (iterator.MoveNext())
        {
            object currentItem = iterator.Current();
            Console.WriteLine($"Item: {currentItem}");
        }

        /*
         * Item: Item 2
         * Item: Item 3
         * Item: Item 4
         * Item: Item 5
         */
    }
}

Define simplified communication between classes.
using System.Collections.Generic;

// Mediator interface declares methods for communication between colleagues.
interface IMediator
{
    void SendMessage(Colleague sender, string message);
}

// Colleague abstract class represents objects that communicate through the mediator.
abstract class Colleague
{
    protected IMediator mediator;

    public Colleague(IMediator mediator)
    {
        this.mediator = mediator;
    }

    public abstract void Send(string message);
    public abstract void Receive(string message);
}

// ConcreteMediator is a concrete implementation of the Mediator interface.
class ConcreteMediator : IMediator
{
    private List<Colleague> colleagues = new List<Colleague>();

    public void AddColleague(Colleague colleague)
    {
        colleagues.Add(colleague);
    }

    public void SendMessage(Colleague sender, string message)
    {
        foreach (var colleague in colleagues)
        {
            if (colleague != sender)
            {
                colleague.Receive(message);
            }
        }
    }
}

// ConcreteColleagueA and ConcreteColleagueB are concrete implementations of the Colleague abstract class.
class ConcreteColleagueA : Colleague
{
    public ConcreteColleagueA(IMediator mediator) : base(mediator) { }

    public override void Send(string message)
    {
        Console.WriteLine("Colleague A sends message: " + message);
        mediator.SendMessage(this, message);
    }

    public override void Receive(string message)
    {
        Console.WriteLine("Colleague A receives message: " + message);
    }
}

class ConcreteColleagueB : Colleague
{
    public ConcreteColleagueB(IMediator mediator) : base(mediator) { }

    public override void Send(string message)
    {
        Console.WriteLine("Colleague B sends message: " + message);
        mediator.SendMessage(this, message);
    }

    public override void Receive(string message)
    {
        Console.WriteLine("Colleague B receives message: " + message);
    }
}

class Program
{
    static void Main(string[] args)
    {
        // Client code creates a mediator and colleagues, then lets them communicate.
        ConcreteMediator mediator = new ConcreteMediator();

        ConcreteColleagueA colleagueA = new ConcreteColleagueA(mediator);
        ConcreteColleagueB colleagueB = new ConcreteColleagueB(mediator);

        mediator.AddColleague(colleagueA);
        mediator.AddColleague(colleagueB);

        colleagueB.Send("Hi from Colleague B!");

         * Colleague A sends message: Hello from Colleague A!
         * Colleague B receives message: Hello from Colleague A!
         * Colleague B sends message: Hi from Colleague B!

A way of notifying change to a number of classes.
using System.Collections.Generic;

// Subject interface declares methods for adding, removing, and notifying observers.
interface ISubject
{
    void AddObserver(IObserver observer);
    void RemoveObserver(IObserver observer);
    void NotifyObservers();
}

// ConcreteSubject class is the subject that maintains a list of observers.
class ConcreteSubject : ISubject
{
    private List<IObserver> observers = new List<IObserver>();
    private string state;

    public string State
    {
        get { return state; }
        set
        {
            state = value;
            NotifyObservers();
        }
    }

    public void AddObserver(IObserver observer)
    {
        observers.Add(observer);
    }

    public void RemoveObserver(IObserver observer)
    {
        observers.Remove(observer);
    }

    public void NotifyObservers()
    {
        foreach (var observer in observers)
        {
            observer.Update(State);
        }
    }
}

// Observer interface declares an Update method to receive updates from the subject.
interface IObserver
{
    void Update(string state);
}

// ConcreteObserver class is a concrete implementation of the Observer interface.
class ConcreteObserver : IObserver
{
    private string name;

    public ConcreteObserver(string name)
    {
        this.name = name;
    }

    public void Update(string state)
    {
        Console.WriteLine($"{name} received update. New state: {state}");
    }
}

class Program
{
    static void Main(string[] args)
    {
        /// Client code creates a subject, observers, and establishes the relationships.
        ConcreteSubject subject = new ConcreteSubject();

        ConcreteObserver observer1 = new ConcreteObserver("Observer 1");
        ConcreteObserver observer2 = new ConcreteObserver("Observer 2");

        subject.AddObserver(observer1);
        subject.AddObserver(observer2);

        // Client code triggers a state change in the subject, and observers are notified.
        subject.State = "New State";

        /*
         * Observer 1 received update. New state: New State
         * Observer 2 received update. New state: New State
         */
    }
}

// Composite
public class Manager : IEmployee
{
    private readonly string name;
    private readonly List<IEmployee> subordinates = new List<IEmployee>();

    public Manager(string name)
    {
        this.name = name;
    }

    public void AddSubordinate(IEmployee subordinate)
    {
        subordinates.Add(subordinate);
    }

    public void DisplayDetails()
    {
        Console.WriteLine($"Manager: {name}");

        foreach (var subordinate in subordinates)
        {
            subordinate.DisplayDetails();
        }
    }
}

class Program
{
    static void Main(string[] args)
    {
        // Creating leaf nodes
        IEmployee developer1 = new IndividualContributor("Alice");
        IEmployee developer2 = new IndividualContributor("Bob");

        // Creating composite node (Manager) and adding leaf nodes (IndividualContributors)
        Manager teamLead = new Manager("Adam");
        teamLead.AddSubordinate(developer1);
        teamLead.AddSubordinate(developer2);

        // Creating another leaf node
        IEmployee manager1 = new Manager("Brad");

        // Adding the manager to the composite node (TeamLead)
        teamLead.AddSubordinate(manager1);

        // Displaying the details, which internally displays details of all subordinates
        teamLead.DisplayDetails();

        /*
         * Individual Contributor: Alice
         * Individual Contributor: Bob
         * Manager: Manager1
        */
    }
}

// Flyweight interface
public interface IFlyweight
{
    void Operation();
}

// Concrete Flyweight
public class ConcreteFlyweight : IFlyweight
{
    private string intrinsicState; // Intrinsic state shared among flyweight objects

    public ConcreteFlyweight(string intrinsicState)
    {
        this.intrinsicState = intrinsicState;
    }

    public void Operation()
    {
        Console.WriteLine($"ConcreteFlyweight: Operation with intrinsic state {intrinsicState}");
    }
}

// Flyweight Factory
public class FlyweightFactory
{
    private readonly Dictionary<string, IFlyweight> flyweights = new();

    public IFlyweight GetFlyweight(string key)
    {
        if (!flyweights.ContainsKey(key))
        {
            flyweights[key] = new ConcreteFlyweight(key);
        }

        return flyweights[key];
    }
}

class Program
{
    static void Main(string[] args)
    {
        FlyweightFactory factory = new();

        IFlyweight flyweight1 = factory.GetFlyweight("A");
        IFlyweight flyweight2 = factory.GetFlyweight("B");
        IFlyweight flyweight3 = factory.GetFlyweight("A");

        // The intrinsic state is shared among flyweights with the same key
        flyweight1.Operation(); // ConcreteFlyweight: Operation with intrinsic state A
        flyweight2.Operation(); // ConcreteFlyweight: Operation with intrinsic state B
        flyweight3.Operation(); // ConcreteFlyweight: Operation with intrinsic state A
    }
}

It's for achieving concurrency, allowing objects to perform actions concurrently while encapsulating the details of how the concurrency is managed. It's often used in scenarios where multiple threads need to interact with an object.
using System;
using System.Collections.Concurrent;
using System.Threading.Tasks;

public class ActiveObject
{
    private readonly BlockingCollection<Func<Task>> methodQueue = new();
    private Task workerTask;

    public ActiveObject()
    {
        workerTask = Task.Run(Run);
    }

    public async Task<int> Add(int a, int b)
    {
        var taskCompletionSource = new TaskCompletionSource<int>();

        methodQueue.Add(async () =>
        {
            int result = a + b;
            taskCompletionSource.SetResult(result);
        });

        return await taskCompletionSource.Task;
    }

    private async Task Run()
    {
        foreach (var method in methodQueue.GetConsumingEnumerable())
        {
            await method.Invoke().ConfigureAwait(false);
        }
    }

    public void Stop()
    {
        methodQueue.CompleteAdding();
        workerTask.Wait(); // Wait for the worker task to complete
    }
}

class Program
{
    static async Task Main()
    {
        ActiveObject activeObject = new();

        Console.WriteLine("Adding 3 + 5");
        int result = await activeObject.Add(3, 5);
        Console.WriteLine($"Result: {result}");

        Console.WriteLine("Adding 10 + 20");
        int result2 = await activeObject.Add(10, 20);
        Console.WriteLine($"Result: {result2}");

        // Stop the active object
        activeObject.Stop();

        /*
         * Adding 3 + 5
         * Result: 8
         * Adding 10 + 20
         * Result: 30
         */
    }
}

public class Singleton

  private static readonly object lockObject = new object();

  private Singleton() { }

  public static Singleton Instance
  {
      get
      {
          // First check without locking (for performance)
          if (instance == null)
          {
              lock (lockObject)
              {
                  // Second check with locking
                  if (instance == null)
                  {
                      Console.WriteLine("Creating Singleton Instance");
                      instance = new Singleton();
                  }
              }
          }

          return instance;
      }
  }

The Producer-Consumer pattern is a classic concurrency pattern where there are two types of threads: producers and consumers. Producers generate data, and consumers consume that data. The pattern is used to safely transfer data from the producer to the consumer, and it often involves a shared data structure like a queue.

    public void Start()
    {
        // Start producer and consumer tasks
        Task.Run(Producer);
        Task.Run(Consumer);
    }

    private void Producer()
    {
        for (int i = 0; i < 10; i++)
        {
            Thread.Sleep(100); // Simulate some work
            int data = i + 1;
            buffer.Add(data);
            Console.WriteLine($"Produced: {data}, Buffer Size: {buffer.Count}");
        }

        buffer.CompleteAdding(); // Indicate that no more items will be added

    {
        foreach (var data in buffer.GetConsumingEnumerable())
        {
            Thread.Sleep(200); // Simulate some work
            Console.WriteLine($"Consumed: {data}, Buffer Size: {buffer.Count}");
        }
    }
}

For controlling access to shared data.
using System.Threading;

public class ReaderWriterLockExample
{
    private readonly ReaderWriterLockSlim rwLock = new();
    private int sharedResource = 0;

    public void Start()
    {
        // Start multiple reader and writer tasks
        for (int i = 0; i < 3; i++)
        {
            Task.Run(Reader);
            Task.Run(Writer);
        }
    }

    private void Reader()
    {
        while (true)
        {
            rwLock.EnterReadLock();
            try
            {
                Console.WriteLine($"Reading shared resource: {sharedResource}");
                Thread.Sleep(200); // Simulate some reading work
            }
            finally
            {
                rwLock.ExitReadLock();
            }

            Thread.Sleep(100); // Introduce delay for better visibility
        }
    }

    private void Writer()
    {
        while (true)
        {
            rwLock.EnterWriteLock();
            try
            {
                int newValue = new Random().Next(1, 100);
                sharedResource = newValue;
                Console.WriteLine($"Writing to shared resource: {newValue}");
                Thread.Sleep(500); // Simulate some writing work
            }
            finally
            {
                rwLock.ExitWriteLock();
            }

            Thread.Sleep(500); // Introduce delay for better visibility
        }
    }
}

class Program
{
    public static void Main()
    {
        ReaderWriterLockExample example = new();
        example.Start();

        Console.ReadLine(); // Allow time for reader and writer tasks to continue running
    }
}

The Thread-Specific Storage (TSS) pattern is a design pattern that allows data to be associated with a specific thread. It's useful in scenarios where you want to maintain separate state or context for each thread. In C#, the ThreadLocal class can be used to implement the Thread-Specific Storage pattern.
using System.Threading;
using System.Threading.Tasks;

public class ThreadSpecificStorageExample
{
    // ThreadLocal instance to store thread-specific data
    private ThreadLocal<int> threadLocalData = new(() => Thread.CurrentThread.ManagedThreadId);

    public void Start()
    {
        // Start multiple tasks accessing thread-local data
        for (int i = 0; i < 5; i++)
        {
            Task.Run(() => AccessThreadLocalData());
        }
    }

    private void AccessThreadLocalData()
    {
        // Accessing thread-local data within a task
        Console.WriteLine($"Thread ID: {Thread.CurrentThread.ManagedThreadId}, ThreadLocal Data: {threadLocalData.Value}");

        // Simulate some work
        Thread.Sleep(200);

        // ThreadLocal data is independent for each thread
        Console.WriteLine($"Thread ID: {Thread.CurrentThread.ManagedThreadId}, Updated ThreadLocal Data: {threadLocalData.Value}");
    }
}


class Program
{
    public static void Main()
    {
        ThreadSpecificStorageExample example = new();
        example.Start();

         * Thread ID: 9, ThreadLocal Data: 9
         * Thread ID: 4, Updated ThreadLocal Data: 4

An object that can be written to only once.
public class WriteOnce<T>
{
  private T value;
  private bool isSet = false;

  public T Value
  {
      get
      {
          if (!isSet)
          {
              throw new InvalidOperationException("Value has not been set.");
          }
          return value;
      }
  }

  public void SetValue(T newValue)
  {
      if (isSet)
      {
          throw new InvalidOperationException("Value can only be set once.");
      }

      value = newValue;
      isSet = true;
  }
}
